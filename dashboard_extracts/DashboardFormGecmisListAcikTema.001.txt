                        if (temizSatir.Contains("var SistemAdi") || temizSatir.Contains("var MySistemAdi"))
                        {
                            int baslangic = temizSatir.IndexOf('"');
                            int bitis = temizSatir.LastIndexOf('"');
                            if (baslangic >= 0 && bitis > baslangic)
                            {
                                alSatKodu = temizSatir.Substring(baslangic + 1, bitis - baslangic - 1);
                            }
                        }
                        
                        if (temizSatir.Contains("var Periyot"))
                        {
                            int baslangic = temizSatir.IndexOf('"');
                            int bitis = temizSatir.LastIndexOf('"');
                            if (baslangic >= 0 && bitis > baslangic)
                            {
                                periyot = temizSatir.Substring(baslangic + 1, bitis - baslangic - 1);
                            }
                        }
                        
                        if (!string.IsNullOrEmpty(alSatKodu) && !string.IsNullOrEmpty(periyot))
                            break;
                    }
                    
                    if (!string.IsNullOrEmpty(alSatKodu))
                    {
                        lblSistemDetay.Text = string.Format("
Periyot: {1}dk", alSatKodu, periyot);
                        lblSistemDetay.ForeColor = Color.FromArgb(0, 120, 215);

                    {
                        lblSistemDetay.Text = "
";

                    }
                }
            }

Hata: " + ex.Message;
