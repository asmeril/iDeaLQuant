Action<DateTime, DateTime, string> gecmisYukle = (baslangic, bitis, filtre) =>
{
    try
    {
        panelGecmis.Controls.Clear();
        
        string robotAdi = txtSistemAdi.Text.Trim();
        string logDosyasi = "Y:\\iDeal\\TradeLogs\\TradeHistory_" + robotAdi + ".txt";
        
        if (!File.Exists(logDosyasi))
        {
            Label uyariLabel = new Label
            {
                Text = "
bulunmuyor.\n\nPozisyon kapatt
necek.",
                Font = new Font("Segoe UI", 12, FontStyle.Bold),
                ForeColor = Color.Orange,
                Location = new Point(20, 50),
                AutoSize = true
            };
            panelGecmis.Controls.Add(uyariLabel);
            
            Label bilgiLabel = new Label
            {
                Text = "
: " + logDosyasi,
                Font = new Font("Consolas", 9),
                ForeColor = Color.Gray,
                Location = new Point(20, 120),
                AutoSize = true
            };
            panelGecmis.Controls.Add(bilgiLabel);
            lblGecmisOzet.Text = "";
            return;
        }

        var satirlar = File.ReadAllLines(logDosyasi);
        var islemler = new List<Dictionary<string, string>>();
        
        foreach (var satir in satirlar)
        {
            if (string.IsNullOrWhiteSpace(satir)) continue;
            
            var parcalar = satir.Split('|');
            if (parcalar.Length >= 9)
            {
                var islem = new Dictionary<string, string>();
                islem.Add("Tarih", parcalar[0].Trim());
                islem.Add("Sembol", parcalar[1].Trim());
                islem.Add("Miktar", parcalar[2].Trim());
                islem.Add("AlisFiyat", parcalar[3].Trim());
                islem.Add("SatisFiyat", parcalar[4].Trim());
                islem.Add("KZ", parcalar[5].Trim());
                islem.Add("KZYuzde", parcalar[6].Trim());
                islem.Add("Gun", parcalar[7].Trim());
                islem.Add("Maliyet", parcalar[8].Trim());
                islemler.Add(islem);
            }
        }

        var filtreliIslemler = islemler.Where(i =>
        {
            DateTime islemTarih;
            if (!DateTime.TryParse(i["Tarih"], out islemTarih)) return false;
            
            bool tarihFiltre = islemTarih >= baslangic && islemTarih <= bitis.AddDays(1);
            
            if (filtre == "Karl
n")
            {
                return islemTarih.Date == DateTime.Now.Date;
            }
            else if (filtre == "Bu Hafta")
            {
                return islemTarih >= DateTime.Now.AddDays(-7);
            }
            else if (filtre == "Bu Ay")
            {
                return islemTarih >= DateTime.Now.AddMonths(-1);
            }
            
            return tarihFiltre;
        }).OrderByDescending(i => i["Tarih"]).ToList();

        if (filtreliIslemler.Count == 0)
        {
            Label bosLabel = new Label
            {
                Text = "
.",
                Font = new Font("Segoe UI", 12, FontStyle.Bold),
                ForeColor = Color.Orange,
                Location = new Point(20, 50),
                AutoSize = true
            };
            panelGecmis.Controls.Add(bosLabel);
            lblGecmisOzet.Text = "";
            return;
        }

        int toplamIslem = filtreliIslemler.Count;
        int karliIslem = 0;
        int zararliIslem = 0;
        double toplamKZ = 0;
        double toplamMaliyet = 0;

        foreach (var islem in filtreliIslemler)
        {
            double kz = 0;
            double maliyet = 0;
            double.TryParse(islem["KZ"], out kz);
            double.TryParse(islem["Maliyet"], out maliyet);
            
            toplamKZ += kz;
            toplamMaliyet += maliyet;
            
            if (kz > 0) karliIslem++;
            else if (kz < 0) zararliIslem++;
        }

        double basariOrani = (toplamIslem > 0) ? (karliIslem * 100.0 / toplamIslem) : 0;
        double toplamKZYuzde = (toplamMaliyet > 0) ? (toplamKZ / toplamMaliyet) * 100 : 0;

        Color ozetRenk = toplamKZ > 0 ? Color.LimeGreen : (toplamKZ < 0 ? Color.OrangeRed : Color.Gray);
        lblGecmisOzet.Text = string.Format("{0} i
",
            toplamIslem, karliIslem, zararliIslem, basariOrani);
        lblGecmisOzet.ForeColor = ozetRenk;

        {
            Location = new Point(10, 10),
            Size = new Size(1150, 80),
            BackColor = Color.FromArgb(40, 40, 40),
            BorderStyle = BorderStyle.FixedSingle
        };

        {
            Text = string.Format("
Toplam K/Z: {0}{1:0.00} ({2}{3:0.00}%)",

                (toplamKZYuzde > 0 ? "+" : ""), toplamKZYuzde),
            Font = new Font("Segoe UI", 14, FontStyle.Bold),
